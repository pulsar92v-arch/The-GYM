let members = JSON.parse(localStorage.getItem("members")) || [];

function addMember() {
const name = document.getElementById("name").value;
const joinDate = new Date(document.getElementById("joinDate").value);
const planDays = parseInt(document.getElementById("plan").value);

if (!name || !joinDate) {
alert("Please fill all fields");
return;
}

const expiry = new Date(joinDate);
expiry.setDate(expiry.getDate() + planDays);

members.push({
name,
expiry: expiry.toISOString()
});

localStorage.setItem("members", JSON.stringify(members));
displayMembers();
}

function displayMembers() {
const list = document.getElementById("memberList");
const alerts = document.getElementById("alerts");

list.innerHTML = "";
alerts.innerHTML = "";

const today = new Date();

members.forEach(m => {
const expDate = new Date(m.expiry);

```
const li = document.createElement("li");
li.innerText = `${m.name} - Expires: ${expDate.toDateString()}`;
list.appendChild(li);

const diffDays = Math.ceil((expDate - today) / (1000 * 60 * 60 * 24));

if (diffDays <= 3) {
  const alertItem = document.createElement("li");
  alertItem.innerText = `⚠️ ${m.name} expires in ${diffDays} day(s)`;
  alerts.appendChild(alertItem);
}
```

});
}

displayMembers();
